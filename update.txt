CREATE TABLE `nicrecord`.`addressbook` ( `uid` BIGINT(20) NOT NULL AUTO_INCREMENT COMMENT 'ID' , `name` VARCHAR(10) NOT NULL COMMENT '姓名' , `sex` VARCHAR(10) NOT NULL COMMENT '性别' , `place` VARCHAR(100) NOT NULL COMMENT '居住地' , `phone` BIGINT(12) NOT NULL COMMENT '手机号' , `qq` BIGINT(12) NOT NULL COMMENT 'qq' , `wechat` VARCHAR(20) NOT NULL COMMENT '微信' , `company` VARCHAR(50) NOT NULL COMMENT '工作单位' , `position` VARCHAR(50) NOT NULL COMMENT '职务' , PRIMARY KEY (`uid`)) ENGINE = MyISAM;
